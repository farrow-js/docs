## Usage

```ts
import { Struct, ID, TypeOf } from "farrow-schema";
import { Validator } from "farrow-schema/validator";

const user = Struct({
  id: ID,
  name: String,
});

const result = Validator.validate(User, {
  id: "foo",
  name: "foo name",
});

if (result.isOk) {
  console.log(result.value);
}

type User = TypeOf<typeof User>;

// 相同类型
type User = {
  id: string;
  name: string;
};
```
